function doPost(e) { try { // 1. 사진 업로드 처리 (FormData 방식 완벽 호환) if (e.parameter.action === 'upload') { var folderId = '1uV3gloGmxupYt-HBKiygfjPePSMNK7JR'; // 신부님 드라이브 폴더 var folder = DriveApp.getFolderById(folderId); // 전달받은 데이터를 파일로 복원하여 드라이브에 저장 var blob = Utilities.newBlob(Utilities.base64Decode(e.parameter.fileData), e.parameter.mimeType, e.parameter.fileName); folder.createFile(blob); return ContentService.createTextOutput("Upload Success"); } // 2. 참석 정보(RSVP) 처리 else { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rowData = [ new Date(), e.parameter.side, e.parameter.name, e.parameter.companion, e.parameter.child, e.parameter.attend, e.parameter.bus ]; sheet.appendRow(rowData); return ContentService.createTextOutput("RSVP Success"); } } catch (error) { return ContentService.createTextOutput("Error: " + error.toString()); } }
